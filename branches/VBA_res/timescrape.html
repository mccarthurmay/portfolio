Time Scrape

            
                Sub TimeScrape()

                Worksheets("ELV").Activate
                
                Do         ' Begin the loop.
                        
                ActiveCell.Offset(1, 0).Select
                Range("E11").Copy
                ActiveCell.PasteSpecial Paste:=xlPasteValues
                Worksheets(ActiveSheet.Index + 1).Select
                
                If ActiveSheet.Index = 395 Then
                    Application.Wait (Now + TimeValue("0:30:00")) ' 30 min
                    Worksheets(3).Select
                ElseIf Hour(Now) <= 9.5 Or Hour(Now) >= 16 Then
                    Application.Wait (Now + TimeValue("0:17:30"))
                End If ''to pause
                
                Loop
                
                
                End Sub
